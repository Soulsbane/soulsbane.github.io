    Struct StructOptions

    Used for the creation of key/value configuration format.

    Template arguments

    struct StructOptions(T);

    Methods

    Name Description
    as Retrieves the value associated with key where T is the designated type to be converted to.
    contains Determines if the key is found in the config file.
    loadFile Loads a config fileName(app.config by default).
    loadString Similar to loadFile but loads and processes the passed string instead.
    save Saves config values to the config file to app.config or the user supplied name.
    set Sets a config value.

    Example

    struct VariedData
{
	string name;
	size_t id;
}

immutable string data =
q{
		name = Paul
		id = 50
};

StructOptions!VariedData options;
options.loadString(data);

assert(options.as("name", "onamae") == "Paul");
assert(options.as("foo", "bar") == "bar");

assert(options.contains("id"));
assert(options.asInteger("id", 10) == 50);
assert(options.asInteger("id") == 50);

assert(options.name == "Paul");

options.name = "Bob";
assert(options.name == "Bob");

options.set("name", "Kyle");
assert(options.name == "Kyle");

options["name"] = "Jim";
assert(options.name == "Jim");

assert(options.as!long("id", 1) == 50); // Can be infered but we'll explicitly send it here.
