    Struct KeyValueConfig

    Handles the processing of config files.

    Methods

    Name Description
    contains Determines if the key is found in the config file. The key can be either its name of in the format of groupName.keyName or just the key name.
    contains Determines if the key is found in the config file.
    containsGroup Determines if the given group exists.
    getGroup Retrieves key/values associated with the group portion of a config file/string.
    getGroups Retrieves an array containing key/values of all groups in the configfile omitting groupless key/values.
    getGroupValue Gets the value associated with the group and key.
    loadFile Loads a config fileName(app.config by default) to be processed.
    loadString Similar to loadFile but loads and processes the passed string instead.
    opIndex Allows config values to be accessed as you would with an associative array.
    opIndexAssign Allows config values to be assigned as you would with an associative array.
    remove Removes a key/value from config file. The key can be either its name of in the format of groupName.keyName or just the keyName.
    remove Removes a key/value from config file. The key can be either its name of in the format of group.keyor just the key.
    removeGroup Removes a group from the config file.
    save Saves config values to the config file.
    set Sets a config value.
    set Sets a config value.

    Example

    string text = "
	aBool=true
	decimal = 3443.443
	number=12071
	#Here is a comment
	sentence=This is a really long sentence to test for a really long value string!
	time=12:04
	[section]
	groupSection=is really cool if this works!
	japan=true
	babymetal=the one
	[another]
	#And another comment!
	world=hello
	japan=false
";

KeyValueConfig config;

immutable bool loaded = config.loadString(text);
assert(loaded, "Failed to load string!");

assert(config.containsGroup("section"));
config.removeGroup("section");
assert(config.containsGroup("section") == false);

assert(config.asBoolean("aBool"));
assert(config.contains("time"));

auto number = config["number"];

assert(number == 12071);
assert(config["decimal"] == 3443.443);

assert(config.contains("another.world"));
assert(config["another.world"] == "hello");
config.remove("another.world");
assert(config.contains("another.world") == false);
assert(config.contains("anothers", "world") == false);

assert(config.contains("number"));
config.remove("number");
assert(config.contains("number") == false);

assert(config["another.japan"] == false);

writeln("Testing getGroup...");

auto group = config.getGroup("another");

foreach(value; group)
{
	writeln(value);
}

writeln();

config.set("aBool", false);
config["aBool"] = true;
assert(config["aBool"].asString == "true");

debug config.save();
debug config.save("custom-config-format.dat");

string noEqualSign = "
	equal=sign
	time=12:04
	This is a really long sentence to test for a really long value string!
";

bool equalSignValue = config.loadString(noEqualSign);
assert(equalSignValue == false);

string invalidGroup = "
	[first]
	equal=sign
	time=12:04
	[second
	another=key value is here
";

bool invalidGroupValue = config.loadString(invalidGroup);
assert(invalidGroupValue == false);
